1. 服务端处理上传与下载的示例:

```csharp
public class AvatarService
{
    [UploadMethod] //必须标明这是个用于客户端上传的方法
    public Task Upload(IAsyncEnumerable<IBlobChunk> chunks, string user)
    {
        //可以前置判断上传权限，抛出异常终止上传
        
        //1. 将上传数据写入文件示例
        var filePath = $"avatar-{user}.jpg";
        await chunks.WriteToFile(filePath);
        //2. 将上传数据写入内存示例
        using var ms = new MemoryStream();
        await chucks.WriteToStream(ms);
        
        //上传方法可以与普通方法一样返回值
    }
    
    [DownloadMethod] //必须标明这是个用于客户端下载的方法
    public Task<Stream> Download(string user)
    {
        //可以前置判断下载权限，抛出异常终止下载
        
        var fileStream = File.OpenRead($"avatar-{user}.jpg");
        //注意:暂只支持返回Stream类型,另不要Dispose返回的Stream,服务端发送完数据后会处理
        return Task.FromResult(fileStream);
    }
    
}
```


2. 客户端处理上传与下载的示例:

```csharp
public class DemoView : View
{
    
    private async void OnUpload()
    {
        using var fileStream = File.OpenRead("avatar.jpg");
        await sys.Services.AvatarService.Upload(fileStream, "Rick");
    }
    
    private async void OnDownload()
    {
        using var ms = new MemoryStream();
        await sys.Services.AvatarService.Download(ms, "Rick");
    }
}
```
