1. 服务模型: 用于向客户端发送事件的测试服务

```csharp
public sealed class TestEventService
{
    public void RaiseEvent(string name)
    {
        var session = RuntimeContext.CurrentSession!;
        var channel = (AppBoxServer.IRemoteChannel)session.Channel;
        
        Task.Run(async () => 
        {
            for(var i = 0; i < 5; i++)
            {
                var result = $"Hello {name} {i + 1}";
                channel.SendServerEvent(888, AnyArgs.Make(result));
                await Task.Delay(500);
            }
        });
        
    }
}
```

2. 视图模型: 用于模拟激发服务端事件并接收事件刷新界面

```csharp
namespace sys.Views;

public sealed class TestEvent : View
{
    public TestEvent()
    {
        Child = new Center()
        {
            Child = new Column
            {
                Spacing = 10,
                Children =
                [
                    new TextInput(_name) { Width = 200 },
                    new Button("Raise Server Event")
                    {
                        OnTap = _ =>
                        {
                            // 通知服务端产生事件
                            sys.Services.TestEventService.RaiseEvent(_name.Value);
                        }
                    },
                    new Text(_message) { FontSize = 50 }
                ]
            }
        };
    }

    private readonly State<string> _name = "Rick";
    private readonly State<string> _message = "Hello World";

    protected override void OnMounted()
    {
        try
        {
            //收到服务端事件的处理
            Channel.AddServerEventHandler(888, args =>
            {
                _message.Value = args.GetString()!;
            });
        }
        catch (Exception ex)
        {
            Notification.Error(ex.Message);
        }
    }

    protected override void OnUnmounted()
    {
        Channel.RemoveServerEventHandler(888);
    }
}
```
